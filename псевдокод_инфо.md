### 7. API Service (api.js)

```pseudocode

    workouts: LOAD_FROM_LOCALSTORAGE("workout_data") || []

FUNCTION delay(ms):
    RETURN Promise that resolves after ms milliseconds

    FUNCTION register(email, password, name):

        RETURN LOAD_FROM_LOCALSTORAGE("current_user")

        AWAIT delay(300)

        result = {
            id: GENERATE_ID(),
            userId: userId,
            ...workoutData,
            completedAt: NOW()

        ADD result to storage.workouts
        SAVE storage.workouts to localStorage
        RETURN result

FUNCTION calculateStreak(workouts):
    IF workouts.length == 0:
        RETURN 0

        IF workoutDate == checkDate:
            INCREMENT streak
            DECREMENT checkDate by 1 day
        ELSE IF workoutDate < checkDate:
            BREAK

    RETURN streak
```
